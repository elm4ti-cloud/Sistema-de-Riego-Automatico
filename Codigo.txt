#include <Servo.h>

Servo servo_10;

void setup()
{
  pinMode(A1, INPUT);
  Serial.begin(9600);
  pinMode(13, OUTPUT);
  servo_10.attach(10, 500, 2500);
}

void loop()
{
  Serial.println(analogRead(A1));
  if (analogRead(A1) < 400) {
    digitalWrite(13, HIGH);
    servo_10.write(180);
  } else {
    digitalWrite(13, LOW);
    servo_10.write(0);
  }
  delay(10);
}
